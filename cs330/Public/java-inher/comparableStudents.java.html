 class Student extends Person
               implements Comparable<Student>
 {
   double gpa;
   String school;

   int compareTo(Object o)
   {
     Student s = (Student)o;
     if (gpa < s.gpa)
       return -1;
     else if (gpa == s.gpa)
       return 0;
     else
       return 1;
   }
}
